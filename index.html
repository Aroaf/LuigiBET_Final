// ============================================================ // Script para Chat Global - LuigiBET 2026 // Guarda y recupera mensajes de la hoja "Hoja 1" // Estructura: Columna A: Hora | Columna B: Usuario | Columna C: Mensaje // ============================================================ function doGet(e) { // Si se recibe el parámetro action=get, devuelve todos los mensajes en formato JSON if (e && e.parameter && e.parameter.action === 'get') { return getMessages(); } return ContentService.createTextOutput('Chat funcionando. Usa ?action=get para obtener mensajes.'); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const { usuario, texto } = data; if (!usuario || !texto) { throw new Error('Faltan datos: usuario y texto son obligatorios'); } // Guardar el mensaje en la hoja const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Hoja 1'); if (!sheet) { throw new Error('La hoja "Hoja 1" no existe. Por favor, crea una hoja con ese nombre.'); } // Asegurar que la hoja tenga los encabezados si está vacía if (sheet.getLastRow() === 0) { sheet.appendRow(['Hora', 'Usuario', 'Mensaje']); } // Agregar el nuevo mensaje con timestamp const now = new Date(); const timestamp = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss'); sheet.appendRow([timestamp, usuario, texto]); return ContentService .createTextOutput(JSON.stringify({ success: true, message: 'Mensaje guardado' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: error.message })) .setMimeType(ContentService.MimeType.JSON); } } function getMessages() { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Hoja 1'); if (!sheet) { throw new Error('La hoja "Hoja 1" no existe.'); } const data = sheet.getDataRange().getValues(); // Si solo tiene la fila de encabezados o está vacía, devolver array vacío if (data.length <= 1) { return ContentService .createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } // Saltar la fila de encabezados (índice 0) const messages = data.slice(1).map(row => ({ timestamp: row[0] || '', usuario: row[1] || '', texto: row[2] || '' })); return ContentService .createTextOutput(JSON.stringify(messages)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ error: error.message })) .setMimeType(ContentService.MimeType.JSON); } }
